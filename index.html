/** * Standard function to call the Gemini 2.5 Flash API * @param {string} userQuery - The user's input message * @param {string} apiKey - The provided API Key */ async function sendMessageToGemini(userQuery, apiKey) { const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`; const payload = { contents: [{ parts: [{ text: userQuery }] }], systemInstruction: { parts: [{ text: "You are a helpful assistant." }] } }; // Implement exponential backoff for reliability for (let i = 0; i < 5; i++) { try { const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) }); const data = await response.json(); if (!response.ok) { // Handle API-specific errors (e.g., Invalid API Key, Rate Limit) const errorMessage = data.error?.message || `HTTP error! status: ${response.status}`; throw new Error(errorMessage); } // Check if candidates exist and contain parts if (data.candidates && data.candidates[0]?.content?.parts?.[0]?.text) { return data.candidates[0].content.parts[0].text; } else { throw new Error("Invalid response format received from API."); } } catch (error) { if (i === 4) throw error; // Wait with increasing delay: 1s, 2s, 4s, 8s await new Promise(resolve => setTimeout(resolve, Math.pow(2, i) * 1000)); } } }
